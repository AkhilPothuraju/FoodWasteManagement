    <script>
        document.getElementById('courierRegistrationForm').addEventListener('submit', function(event) {
            event.preventDefault();
            const courierName = document.getElementById('courierName').value;
            const mobileNumber = document.getElementById('mobileNumber').value;
            const password = document.getElementById('password').value;

            // Save the courier details in localStorage
            let courierAccounts = JSON.parse(localStorage.getItem('courierAccounts')) || [];
            courierAccounts.push({ courierName, mobileNumber, password });
            localStorage.setItem('courierAccounts', JSON.stringify(courierAccounts));

            // Redirect to the success page
            window.location.href = 'success.html';
        });
    </script>
