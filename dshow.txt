    <script>
        const donationData = JSON.parse(localStorage.getItem('donationEntries')) || [];

        if (donationData.length > 0) {
            const lastDonation = donationData[donationData.length - 1];
            document.getElementById('foodnameDisplay').textContent = lastDonation.foodname;
            document.getElementById('mealDisplay').textContent = lastDonation.meal;
            document.getElementById('categoryDisplay').textContent = lastDonation.category;
            document.getElementById('quantityDisplay').textContent = lastDonation.quantity;
            document.getElementById('nameDisplay').textContent = lastDonation.name;
            document.getElementById('phonenoDisplay').textContent = lastDonation.phoneno;
            document.getElementById('districtDisplay').textContent = lastDonation.district;
            document.getElementById('addressDisplay').textContent = lastDonation.address;
        } else {
            alert("No donation data available.");
        }

        document.getElementById('acceptOrderButton').addEventListener('click', function() {
            if (donationData.length === 0) {
                alert("No donation to accept.");
                return;
            }

            const acceptedDonation = donationData[donationData.length - 1];
            const courierDetails = JSON.parse(localStorage.getItem('currentCourier')) || {};
            
            if (!courierDetails.courierName) {
                alert("Courier details not found.");
                return;
            }

            const deliveryDetails = {
                foodName: acceptedDonation.foodname,
                mealType: acceptedDonation.meal,
                category: acceptedDonation.category,
                quantity: acceptedDonation.quantity,
                name: acceptedDonation.name,
                phoneNo: acceptedDonation.phoneno,
                district: acceptedDonation.district,
                address: acceptedDonation.address,
                courierName: courierDetails.courierName,
            };

            const acceptedDeliveries = JSON.parse(localStorage.getItem('acceptedDeliveryEntries')) || [];
            acceptedDeliveries.push(deliveryDetails);
            localStorage.setItem('acceptedDeliveryEntries', JSON.stringify(acceptedDeliveries));

            alert("Order Accepted! Delivery details saved.");
            window.location.href = ''; // Change to your desired page
        });
    </script>
